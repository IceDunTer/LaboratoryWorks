#define _CRT_SECURE_NO_DEPRECATE
#include <stdio.h>
#include <locale.h>
#include <stdlib.h>
#include <string.h>
#define NMAX 100
#define TYPE char*

typedef struct Stack {
    int top;
    TYPE data[NMAX];
} stack;

int isempty(stack* s) {
    return s->top == -1;
}

int isfull(stack* s) {
    return s->top == NMAX - 1;
}

int push(stack* s, TYPE a) {
    if (!isfull(s)) {
        s->data[++s->top] = a;
        return 1;
    }
    printf("Стек переполнен\n");
    return 0;
}

TYPE pop(stack* s) {
    if (!isempty(s)) {
        return s->data[s->top--];
    }
    printf("Стек пуст\n");
    return NULL;
}

void reverse(char* line, stack* s) {
    char* token = strtok(line, " ");
    while (token != NULL) {
        push(s, token);
        token = strtok(NULL, " ");
    }

    printf("Обратный порядок слов: ");
    while (!isempty(s)) {
        printf("%s ", pop(s));
    }
    printf("\n");
}

int main() {
    setlocale(LC_ALL, "RUS");
    FILE* input_file = fopen("input.txt", "r");
    FILE* output_file = fopen("output.txt", "w");
    if (input_file == NULL || output_file == NULL) {
        printf("Ошибка открытия файла\n");
        return 1;
    }

    stack s = { .top = -1 };
    char line[1024];

    while (fgets(line, sizeof(line), input_file)) {
        reverse(line, &s);
        fprintf(output_file, "%s\n", line);
    }

    fclose(input_file);
    fclose(output_file);
}
