#define _CRT_SECURE_NO_DEPRECATE
#include <stdio.h>
#include <locale.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>
#include <ctype.h>
#include <time.h>
#define NMAX 100
#define TYPE char
#define FTYPE "%3c,"

typedef struct Stack {
    int   top;
    TYPE  data[NMAX];
} stack;

int isempty(stack* s) {
    return s->top == 0;
}

int isfull(stack* s) {
    return s->top == NMAX - 1;
}

int push(stack* s, TYPE a) {
    if (!isfull(s)) {
        s->data[++s->top] = a;
        return 1;
    }
    else {
        printf("Стек переполнен\n");
        return 0;
    }
}

TYPE pop(stack* s) {
    if (!isempty(s)) {
        return s->data[s->top--];
    }
    else {
        printf("Стек пуст\n");
        return '\0';
    }
}

TYPE top(stack* s) {
    if (!isempty(s)) {
        return s->data[s->top];
    }
    return '\0';
}

void display(stack* s) {
    for (int i = s->top; i >= 0; i--) {
        printf(FTYPE, s->data[i]);
    }
}

int main() {
    stack z = { .top = -1 };
    char s[100];
    setlocale(LC_ALL, "RUS");
    printf("Введите строку\n");
    scanf("%s", s);
    for (int i = 0; s[i] != '\0'; i++) {
        push(&z, s[i]);
    }
    display(&z);
    return 0;
}
