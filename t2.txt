#define _CRT_SECURE_NO_DEPRECATE
#include <stdio.h>
#include <locale.h>
#include <stdlib.h>
#include <math.h>
#include <string.h>
#include <ctype.h>
#include <time.h>
#define NMAX 100
#define TYPE char
#define FTYPE "%3c,"

struct  Node {
	TYPE data;
	struct Node* next;
};

typedef struct {
	struct  Node* queue;
	struct  Node* front;
	struct  Node* back;
} queue_list;
void enQueue(queue_list* que, TYPE element) {
	struct Node new;
	if (que->back != NULL) {
		que->back->next = &new;
		que->back = &new;
	}
	else {
		que->back = &new;
		que->front = &new;
	}
}
TYPE deQueue(queue_list* que) {
	TYPE elem;
	if (que->front == NULL) {
		printf("Ошбика");
		return 0;
	}
	else {
		TYPE elem = que->front->data;
		que->front = que->front->next;
	}
	if (que->front == NULL) {
		que->back = NULL;
	}
	return elem;
}
void main() {
	setlocale(LC_ALL, "RUS");
	queue_list str;
	str.back = str.front = NULL;
	char temp[100];
	int cnt = 0;
	puts("ВВЕдите строку");
	scanf("%s", temp);
	for (int i = 0; (i < 100) && (temp[i] != '\n'); i++) {
		enQueue(&str, temp[i]);
		cnt += 1;
	}
	for (int i = cnt; i >= 0; i--) printf(FTYPE, deQueue(&str));
}
